# commands

```
#package manager
opkg

#read logs
logread

#wireless settings
/etc/config/wireless
/etc/config/network

#list available wireless devices
##phy0 will be wlan0
iw list

#scan for wireless networks
iw wlan0 scan
#or
ifconfig wlan0 down
iw phy phy0 interface add scan0 type station
ifconfig scan0 up
iwlist scan0 scan
iw dev scan0 del
ifconfig wlan0 up
killall -HUP hostapd

#reset wifi
wifi down && sleep 5 && wifi
```

# minimal interface configuration

```
#/etc/config/network
config 'interface' 'wan'
    option 'protp' 'dhcp'
    option 'ifname' 'eth0.1'
```

# install gui

```
opkg install luci-ssl
/etc/init.d/uhttpd restart
```
