    arrow-up

# delete one character of last command

    CTRL-h

# delete whole inputline

    CTRL-u

# pause output

    CTRL-s

# restart paused output

    CTRL-q

# send end of input signal

    CTRL-d

# cancel

    CTRL-c

# reset

    CTRL-j

# stop job

    CTRL-z

    jobs

# bring job back

    fg [$numberOfJobFromList]

# continue job in background

    bg [$numberOfJobFromList]

    (commandOne; commandTwo);
