# kickstart

```typescript
#define variables
let name: string = "foobar";
let age: number = 13;
let isCool: boolean = false;

#interface
##define
interface Foo
{
    firstProperty: string;
    secondProperty: string;
}
##usage
function bar(foo : Foo):string {
    return "There is no " + foo.firstProperty + " without a " + foo.secondProperty;
}
```
